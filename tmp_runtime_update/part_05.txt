    import ctypes.wintypes

    user32 = ctypes.windll.user32
    _hotkey_tid = ctypes.windll.kernel32.GetCurrentThreadId()
    if not user32.RegisterHotKey(None, 0xD35, 0x0001 | 0x0002 | 0x4000, ord("C")):
        log(f"⚠️ No se pudo registrar {HOTKEY}")
        return
    try:
        msg = ctypes.wintypes.MSG()
        while not STOP.is_set() and user32.GetMessageW(ctypes.byref(msg), None, 0, 0) > 0:
            if msg.message == 0x0312 and msg.wParam == 0xD35:
                toggle(icon)
    finally:
        user32.UnregisterHotKey(None, 0xD35)


def calibration_window():
    selected = []
    root = Tk()
    root.title(f"Calibrando lector QR - v{VERSION}")
    root.configure(bg=core.COLOR_BG)
    root.protocol("WM_DELETE_WINDOW", lambda: shutdown("Se cerró la calibración"))

    label = ttk.Label(root, text="Inicializando...", font=("Segoe UI", 12))
    label.pack(padx=30, pady=30)

    def status(text):
        try:
            root.after(0, lambda: label.config(text=text))
        except Exception:
            pass

    def worker():
        try:
            last = load_last_com()
            if last and port_responds(last, status, 3):
                selected.append(last)
            elif not STOP.is_set():
                port = find_reader(status)
                if port:
                    save_last_com(port)
                    selected.append(port)
        finally:
            try:
                root.after(0, root.destroy)
            except Exception:
                pass

    threading.Thread(target=worker, name="DMSCalibration", daemon=True).start()
    root.mainloop()
    return selected[0] if selected else None


def detect_reader():
    while not STOP.is_set():
        port = calibration_window()
        if port:
            return port
        if STOP.is_set():
            return None

        dialog = Tk()
        dialog.withdraw()
        dialog.attributes("-topmost", True)
        retry = messagebox.askretrycancel(
            "No se detectó lector",
            "No se detectó el lector QR.\n\n¿Intentar nuevamente?",
            parent=dialog,
        )
        dialog.destroy()
        if not retry:
            shutdown("Se canceló la detección del lector")
    return None


def open_selector(icon, item=None):
    if STOP.is_set() or not _selector_lock.acquire(blocking=False):
        return

    def worker():
        try:
            root = Tk()
            root.withdraw()
            ConfigSelector(root)
            root.destroy()
        except Exception as exc:
            log(f"⚠️ Error abriendo selector: {exc}")
        finally:
            _selector_lock.release()

    threading.Thread(target=worker, name="DMSConfigSelector", daemon=True).start()
