    handle = kernel32.CreateMutexW(None, False, MUTEX_NAME)
    exists = ctypes.get_last_error() == ERROR_ALREADY_EXISTS
    if not exists:
        _mutex_handle = handle
        return True

    kernel32.CloseHandle(handle)
    answer = ctypes.windll.user32.MessageBoxW(
        0,
        "Ya existe una sesión del lector abierta.\n\n"
        "¿Desea detenerla e iniciar esta nueva sesión?",
        "Lector de Cédulas DMS",
        0x00000004 | 0x00000020 | 0x00040000,
    )
    if answer != 6:
        return False

    _kill_previous_instances()
    deadline = time.monotonic() + 8
    while time.monotonic() < deadline:
        ctypes.set_last_error(0)
        handle = kernel32.CreateMutexW(None, False, MUTEX_NAME)
        if ctypes.get_last_error() != ERROR_ALREADY_EXISTS:
            _mutex_handle = handle
            return True
        kernel32.CloseHandle(handle)
        time.sleep(0.2)

    ctypes.windll.user32.MessageBoxW(
        0,
        "No fue posible detener la sesión anterior.\n"
        "Cierre el proceso desde el Administrador de tareas e intente nuevamente.",
        "No se pudo iniciar",
        0x10 | 0x00040000,
    )
    return False


def _session_watch():
    """Cierra esta copia cuando Windows activa otro usuario/sesión."""
    if os.name != "nt":
        return
    try:
        pid = ctypes.windll.kernel32.GetCurrentProcessId()
        own = ctypes.c_ulong()
        ctypes.windll.kernel32.ProcessIdToSessionId(pid, ctypes.byref(own))
        mismatches = 0
        while not STOP.wait(1.0):
            active = int(ctypes.windll.kernel32.WTSGetActiveConsoleSessionId())
            if active != 0xFFFFFFFF and active != own.value:
                mismatches += 1
            else:
                mismatches = 0
            if mismatches >= 2:
                shutdown("Windows cambió de usuario o de sesión")
                return
    except Exception as exc:
        log(f"⚠️ No se pudo supervisar la sesión de Windows: {exc}")


def _wait(seconds):
    return STOP.wait(max(0.0, seconds))


def wait_serial_silent(serial_port, quiet_seconds=0.35, timeout=2.0):
    start = time.monotonic()
    quiet = time.monotonic()
    while not STOP.is_set() and time.monotonic() - start < timeout:
        try:
            waiting = serial_port.in_waiting
            if waiting:
                serial_port.read(waiting)
                quiet = time.monotonic()
            elif time.monotonic() - quiet >= quiet_seconds:
                return
        except Exception:
            return
        if _wait(SERIAL_POLL):
            return


def read_serial_buffer(serial_port, timeout_total=3.0, max_bytes=None, idle_seconds=None):
    max_bytes = max_bytes or core.READ_MAX_BYTES
    idle_seconds = idle_seconds or core.READ_IDLE_SECONDS
    data = bytearray()
    start = time.monotonic()
    last = None

    while not STOP.is_set() and time.monotonic() - start < timeout_total:
        try:
            waiting = serial_port.in_waiting
            if waiting:
                chunk = serial_port.read(waiting)
                if chunk:
                    data.extend(chunk)
                    last = time.monotonic()
                    if len(data) > max_bytes:
                        break
            elif data and last and time.monotonic() - last >= idle_seconds:
                if _wait(0.06):
                    break
                if not serial_port.in_waiting:
                    break
        except Exception:
            break
        if _wait(SERIAL_POLL):
            break
    return bytes(data)


def port_responds(port, callback=None, seconds=3):
    if not port or STOP.is_set():
        return False
    try:
        if callback:
            callback(f"🔁 Probando último puerto guardado: {port}... pase la cédula")
        with core.serial.Serial(port, 9600, timeout=0.25) as serial_port:
            serial_port.reset_input_buffer()
            data = bytearray()
            start = time.monotonic()
            while not STOP.is_set() and time.monotonic() - start < seconds:
                waiting = serial_port.in_waiting
                if waiting:
                    data.extend(serial_port.read(waiting))
                    if len(data) >= 100 or data.endswith(b"\r\n"):
                        if callback:
                            callback(f"✅ {port} respondió correctamente")
                        return True
                if _wait(SERIAL_POLL):
                    break
